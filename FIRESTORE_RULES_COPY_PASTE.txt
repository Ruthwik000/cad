rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users collection
    match /users/{userId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == userId;
      allow update: if request.auth != null && request.auth.uid == userId;
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // Sessions collection - with collaborative sharing support
    match /sessions/{sessionId} {
      // Allow read if:
      // 1. User is authenticated and owns the session, OR
      // 2. Session is marked as shared (for collaborative links)
      allow read: if request.auth != null && 
                     (resource.data.userId == request.auth.uid ||
                      resource.data.isShared == true);
      
      // Allow create if user is authenticated and sets themselves as owner
      allow create: if request.auth != null && 
                       request.resource.data.userId == request.auth.uid;
      
      // Allow update if:
      // 1. User owns the session, OR
      // 2. Session is shared and user is authenticated (for collaboration)
      allow update: if request.auth != null && 
                       (resource.data.userId == request.auth.uid ||
                        resource.data.isShared == true);
      
      // Allow delete only if user owns the session
      allow delete: if request.auth != null && 
                       resource.data.userId == request.auth.uid;
    }
  }
}

IMPORTANT: CREATE COMPOSITE INDEX

After deploying the rules, you MUST create a composite index for queries to work.

Method 1 - Automatic (Recommended):
1. Try to load your sessions in the app
2. Check the browser console for an error with a link
3. Click the link to automatically create the index

Method 2 - Manual:
1. Go to: Firebase Console > Firestore Database > Indexes
2. Click "Create Index"
3. Collection ID: sessions
4. Add field: userId (Ascending)
5. Add field: updatedAt (Descending)
6. Query scope: Collection
7. Click "Create Index"
8. Wait for index to build (usually 1-2 minutes)

Without this index, getUserSessions() will fail with an error like:
"The query requires an index"
